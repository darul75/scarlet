// Try Scarlet Now :) // A simple javascript object with property function MyClass() { var self = this; // A sample property self.property = "anyPropertyValue"; // A sample method self.method = function(param1, param2){ console.log("MyClass::method(param1='"+param1+"', param2='"+param2+"')"); return 0; }; console.log("MyClass::ctor"); } // Instantiate Scarlet var scarlet = new Scarlet(); // Create prototype proxy var MyClassProxy = scarlet .intercept(MyClass, scarlet.PROTOTYPE) .using(function(info, method, args){ var result = method.call(this, info, method, args); console.log("Interceptor::result='"+result+"'"); return result; }).proxy(); // Instantiate MyClass var instance = new MyClassProxy(); // Set property instance.property = "anyOtherValue"; // Get property console.log(instance.property); // Execute method var result = instance.method("param1", "param2"); console.log(result);
