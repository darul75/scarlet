// Try Scarlet Now :) (Click Here && WIN=Ctrl-X, MAC=Cmd+X to execute) // A simple javascript object function MyClass() { this.anyProperty = "oldValue"; this.anyMethod = function(param1, param2){ return 0; }; } // Instantiate Scarlet var scarlet = new Scarlet(); // Create Interceptor var interceptor = function(info, method, args){ var result = method.call(this, info, method, args); var query = scarlet.interceptQuery(info, method, args, result); query.traceTo(print); return result; }; // Create proxy var MyClassProxy = scarlet .intercept(MyClass, scarlet.PROTOTYPE) .using(interceptor) .proxy(); // Instantiate MyClass var instance = new MyClassProxy(); // Set property instance.anyProperty = "newValue"; // Get property var anyResult = instance.anyProperty; // Execute method var result = instance.anyMethod("param1Val", "param2Val");
